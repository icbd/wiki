# 二分法

```ruby

# @param {Integer} x
# @return {Integer}
def my_sqrt(x)
  return 0 if x == 0
  return 1 if x < 4

  mid = x / 2
  loop do
    if mid * mid <= x and (mid + 1) * (mid + 1) > x
      return mid
    elsif mid * mid > x
      mid = (1 + mid) / 2
    else # mid * mid < x
      last_mid = 2 * mid - 1
      mid = (mid + last_mid) / 2
    end
  end

  mid
end

# p my_sqrt 100
```
